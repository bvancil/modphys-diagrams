  \vspace{-3cm}
    \vfill\normalsize
    \begin{block}{Preparing your whiteboards}
      \begin{rulelist}
        \item List how you want students to present an experiment.
        \item Do you want CER like \url{http://alwaysformative.blogspot.com/2012/04/claim-evidence-reasoning.html}?
        \item List what things they should include in their presentation.  Build in some sort of the model-defense protocol below.
      \end{rulelist}
    \end{block}
    \begin{block}{Model defense protocol}
      \begin{rulelist}
	\item One person explains the group's approach to the problem.
	\item Everyone scrutinizes the work silently for two minutes.
	\item Ask questions---don't make statements---to \emph{lead} the presenter to realize inconsistency or errors in the work.
	\item After all groups have gone, erase and clean the whiteboards.
      \end{rulelist}
    \end{block}
