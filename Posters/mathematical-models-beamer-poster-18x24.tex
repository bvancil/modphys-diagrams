    \normalsize Mathematical model with example & \normalsize Kind of change & \normalsize Graph shape  \\ \midrule[.1em] \addlinespace
    
    \formatmm{Linear model}\ \ \  $\yy=A\xx+B$ & 
    Rate of change is constant. &
    \imagetop{\begin{tikzpicture}[scale=\plotscale,domain=0:4,line width=\plotline,smooth]
    \draw[color=plot] plot (\x,.6*\x+1);
    \draw[<->] (0,4) -- (0,0) -- (4,0); 
    \end{tikzpicture}} 
    \\ \addlinespace \midrule \addlinespace
    
    \formatmm{Quadratic model}\ \ \  $\yy=A\xx^{2}+B\xx+C$ & 
    Change in rate of change is constant. &
    \imagetop{\begin{tikzpicture}[scale=\plotscale,domain=0:4,line width=\plotline,smooth,samples=40]
    \draw[color=plot] plot (\x,{4-0.7*(\x-2)*(\x-2)});
    \draw[<->] (0,4) -- (0,0) -- (4,0); 
    \end{tikzpicture}}    
    \\ \addlinespace \midrule \addlinespace

    \formatmm{Exponential model}\ \ \  $\yy=Ab^{\xx}$ or $\yy=Ae^{\frac{\xx}{\xi}}$ & 
    Rate of change is proportional to amount. &
    \imagetop{\begin{tikzpicture}[scale=\plotscale,domain=0:4,line width=\plotline,smooth,samples=40]
    \draw[color=plot] plot (\x,{pow(pow(4,.25),\x)});
    \draw[<->] (0,4) -- (0,0) -- (4,0); 
    \end{tikzpicture}}    
    \\ \addlinespace \midrule \addlinespace
    
    \formatmm{Harmonic model}\ \ \  $\yy=A\cos\left(k\xx+\phi\right)$ or $\yy=A\sin\left(k\xx+\phi'\right)$ or $\yy=\mathfrak{Re}\left\{Ae^{i(k\xx+\phi)}\right\}$& 
    Change in rate of change is proportional to amount. &
    \imagetop{\begin{tikzpicture}[scale=\plotscale,domain=0:4,line width=\plotline,smooth,samples=40]
    \draw[color=plot] plot (\x,{2*cos((\x*6.28-1)r)}); 
    \draw[->] (0,-2) -- (0,2);
    \draw[->] (0,0) -- (4,0);
    \end{tikzpicture}}    
    \\ \addlinespace  \midrule \addlinespace

    \formatmm{Sudden change model}\ \ \  $\yy=A\ \theta(\xx-x_{0})$ & 
    Change is finite and instantaneous. &
    \imagetop{\begin{tikzpicture}[scale=\plotscale,domain=0:4,line width=\plotline,smooth,samples=40]
    \draw[color=plot, domain=0:2] plot (\x,1);
    \draw[color=plot, domain=2:4] plot (\x,3);
    \draw[<->] (0,4) -- (0,0) -- (4,0); 
    \end{tikzpicture}}    
    \\ \addlinespace
